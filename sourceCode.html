// this structure defines all the variables and events of your control interface

  uint8_t sel1; // =0 if select position A, =1 if position B, =2 if position C, ...

void setup() 
{
  RemoteXY_Init (); 
  
  pinMode(8,OUTPUT);  //lights
  pinMode(9,OUTPUT);  //engine
  pinMode(10,OUTPUT); //engine
  
  Serial.begin(9600);
}
int val;
void lights()
{
  val = analogRead(0);
  if(RemoteXY.sel1 == 0)
  {
      digitalWrite(8,LOW);
    }
  
  else if(RemoteXY.sel1 == 1)
  {
      digitalWrite(8,HIGH);
    }
    
  else
  {
      if(val>=20) //nastaviť citlivosť
      {
        if(Lock)
        {
          clap++;
          if(clap==2){
            Lock = false;
            digitalWrite(8,HIGH);
          }
        }
        else if(!Lock)
        {
          clap--;
          delay(100);
          if(clap==0)
          {
              Lock = true;
              digitalWrite(8,LOW);
              clap=0;
            }
           }
          }
      Serial.println(val,DEC);//print the sound value to serial
      delay(100);
    }
  }

void blinds()
{
    if(RemoteXY.downBtn1 == 1)
    {
        digitalWrite(9,LOW);
        digitalWrite(10,HIGH);

      }
    else if(RemoteXY.upBtn1 == 1)
    {
        digitalWrite(9,HIGH);
        digitalWrite(10,LOW);

      }
    else
    {
        digitalWrite(9,LOW);
        digitalWrite(10,LOW);
      }
  }

void loop() 
{ 
  RemoteXY_Handler ();
  lights();
  blinds();
}
