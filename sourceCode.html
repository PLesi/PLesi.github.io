const unsigned long eventInterval = 300000;
bool ohrev = true;

void loop()
{
  RemoteXY_Handler ();
  
  RemoteXY.led_3 = 1;
  RemoteXY.led_2 = 0;
  RemoteXY.led_1 = 0;
  if (ohrev) {
    RemoteXY.led_2 = 0;
    RemoteXY.led_1 = 1;
    RemoteXY.led_3 = 0;
    

    if (RemoteXY.pushSwitch_1 == 1) {
      digitalWrite(11,LOW);
      unsigned long currentTime = millis();
      int interval = currentTime % 3000;
      
        if (interval == 0) {

        RemoteXY.circularbar_1 += 1;
        delay(1);
      }
      if (currentTime - previousTime >= eventInterval) {
          RemoteXY.led_3 = 0;
          RemoteXY.led_2 = 1;
          RemoteXY.led_1 = 0;
          digitalWrite(11,HIGH);
          ohrev = false;
          delay(1);

      }
    }
  }
  if(RemoteXY.rst == 1){
          RemoteXY.led_3 = 0;
          RemoteXY.led_2 = 0;
          RemoteXY.led_1 = 1;
          ohrev = true;
          digitalWrite(11,HIGH);
          RemoteXY.circularbar_1 = 0;
    }
}
